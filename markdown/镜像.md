; cli configs
metrics-registry = "https://registry.npm.taobao.org/"
scope = ""
user-agent = "npm/6.2.0 node/v9.7.1 win32 x64"

; globalconfig D:\Apps\Node\etc\npmrc

; node bin location = D:\Apps\Node\node.exe
; cwd = D:\Apps\Node
